124 private int DisplayAngleSelection(int playerX, int playerY, bool isPlayer1)
125 {
126 int[][] points;
127
128 if (isPlayer1)
129 {
130 // Sélection d'angle pour le joueur 1 (tir vers la droite)
131 points = new int[][]
132 {
133 new int[] {playerX + 2, playerY - 3}, // Point 1 (70°)
134 new int[] {playerX + 4, playerY - 2}, // Point 2 (45°)
135 new int[] {playerX + 6, playerY - 1}, // Point 3 (35°)
136 new int[] {playerX + 8, playerY}, // Point 4 (25°)
137 new int[] {playerX + 10, playerY + 1} // Point 5 (15°)
138 };
139 }
140 else
141 {
142 // Sélection d'angle pour le joueur 2 (tir vers la gauche)
143 points = new int[][]
144 {
145 new int[] {playerX - 2, playerY - 3}, // Point 1 (70°)
146 new int[] {playerX - 4, playerY - 2}, // Point 2 (45°)
147 new int[] {playerX - 6, playerY - 1}, // Point 3 (35°)
148 new int[] {playerX - 8, playerY}, // Point 4 (25°)
149 new int[] {playerX - 10, playerY + 1} // Point 5 (15°)
150 };
151 }
152
153 int[] angles = { 70, 45, 35, 25, 15 };
154 int selectedIndex = 0;
155
156 // Boucle pour afficher la sélection de l'angle
157 while (!Console.KeyAvailable || Console.ReadKey(true).Key != ConsoleKey.Spacebar)
158 {
159 // Effacer les anciens points
160 foreach (var point in points)
161 {
162 Console.SetCursorPosition(point[0], point[1]);
163 Console.Write(' '); // Effacer le point
164 }
165
166 // Afficher les nouveaux points
167 for (int i = 0; i < points.Length; i++)
168 {
169 Console.SetCursorPosition(points[i][0], points[i][1]);
170 if (i == selectedIndex)
171 {
172 Console.Write('O'); // Point sélectionné
173 Console.SetCursorPosition(points[i][0] + (isPlayer1 ? 2 : -2), points[i][1]);
174 }
175 else
176 {
177 Console.Write('*'); // Point non sélectionné
178 }
179 }
180
181 // Changer la sélection de l'angle (clignotement)
182 selectedIndex = (selectedIndex + 1) % points.Length;
183 System.Threading.Thread.Sleep(200); // Délai pour le clignotement
184 }
185
186 return angles[selectedIndex];
187 }
